import { Component, OnInit, Input, Output, EventEmitter} from '@angular/core';
import { Issue } from 'src/app/models/issue';

@Component({
  selector: 'app-card',
  templateUrl: './card.component.html',
  styleUrls: ['./card.component.scss']
})

export class CardComponent{

 @Input() issue: Issue = new Issue;
 @Output() deleteClicked : EventEmitter<string> = new EventEmitter(); 
 @Output() moreClicked : EventEmitter<string> = new EventEmitter(); 
 @Output() cardClicked : EventEmitter<string> = new EventEmitter<string>();

  constructor() {}

  onCardClicked(id) : void{
    this.cardClicked.emit(id);
  }

  onMoreClicked(id) : void {
    this.moreClicked.emit(id);
  }

  onDeleteClicked(id) : void {
    this.deleteClicked.emit(id);
  }

}

                </div>

                <p> created : {{issue.createdAt | date:'mediumDate' }} </p>
                <br>
                <p [ngClass]="{'state-new' : issue.state === 'new',
                            'state-inprogress' : issue.state === 'inProgress',
                            'state-rejected' : issue.state === 'rejected',
                            'state-resolved' : issue.state === 'resolved'}">
                    {{issue.state}}
                </p>
                <button mat-raised-button color="secondary" class="deleteButton" (click)="onDeleteClicked(IdHolder.id)">
                    <mat-icon>delete_outline</mat-icon>
                </button>
                <button mat-raised-button color="primary" class="moreButton" (click)="onMoreClicked(IdHolder.id)">more</button>
            </div>
        </div>
    </mat-card>
</div>
